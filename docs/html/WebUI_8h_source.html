660  void handleWebSocketEvent(AsyncWebSocketClient* client, AwsEventType type,
661  void* arg, uint8_t* data, size_t len) {
662  if (!client) return;
663 
664  switch (type) {
665  case WS_EVT_CONNECT:
666  DLOG_I(LOG_CORE, "[WebUI] WebSocket client connected: %u", client->id());
667  break;
668 
669  case WS_EVT_DISCONNECT:
670  DLOG_I(LOG_CORE, "[WebUI] WebSocket client disconnected: %u", client->id());
671  break;
672 
673  case WS_EVT_DATA: {
674  if (!arg || !data || len == 0 || len > 512) break;
675 
676  AwsFrameInfo* info = (AwsFrameInfo*)arg;
677  if (info->final && info->index == 0 && info->len == len &&
678  info->opcode == WS_TEXT && len < 256) {
679 
680  String message = String((char*)data, len);
681  handleWebSocketMessage(client, message);
682  }
683  break;
684  }
685 
686  case WS_EVT_ERROR:
687  break;
688  }
689  }
