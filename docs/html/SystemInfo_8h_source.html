88  float calculateCpuLoad() {
89  // Heuristic CPU load estimation based on heap allocation activity.
90  // On ESP32 Arduino, direct CPU usage isn't available without special FreeRTOS config.
91  // We therefore estimate activity over time and smooth it with an EMA.
92  unsigned long currentTime = millis();
93  uint32_t currentHeap = ESP.getFreeHeap();
94 
95  if (lastHeapCheck > 0) {
96  unsigned long dt = currentTime - lastHeapCheck;
97  if (dt > 0) {
98  // Heap activity in KB per second
99  float heapDiffKB = fabsf((float)((int32_t)currentHeap - (int32_t)lastHeapValue)) / 1024.0f;
100  float activityPerSec = heapDiffKB * (1000.0f / (float)dt);
101 
102  // Map activity to an arbitrary 0-100 range
103  // Tuned scale: 10 KB/s ~ 100% load (cap at 100)
104  float instantLoad = constrain(activityPerSec * 10.0f, 0.0f, 100.0f);
105 
106  // Exponential moving average for stability
107  const float alpha = 0.3f; // smoothing factor
108  cpuLoadEma = (alpha * instantLoad) + ((1.0f - alpha) * cpuLoadEma);
109  metrics.cpuLoad = constrain(cpuLoadEma, 0.0f, 100.0f);
110  }
111  }
112 
113  lastHeapCheck = currentTime;
114  lastHeapValue = currentHeap;
115  return metrics.cpuLoad;
