149  void poll(size_t maxPerPoll = 8) {

153  queue.pop();
154  processed++;

159  if (qe.topic.length() > 0) {

162  if (itT != topicSubscriptions.end()) {

164  for (const auto& sub : handlers) {

167  }

227  static bool matchesWildcard(const String& concrete, const String& pattern) {
228  int star = pattern.indexOf('*');
229  if (star < 0) return false; // not a wildcard pattern
230  // Allow only prefix+"*" patterns for simplicity
231  String prefix = pattern.substring(0, star);
232  if (star != (int)pattern.length() - 1) {
233  // If pattern has chars after '*', require full match (very simple contains)
234  String suffix = pattern.substring(star + 1);
235  return concrete.startsWith(prefix) && concrete.endsWith(suffix);
236  }
237  return concrete.startsWith(prefix);
238  }
