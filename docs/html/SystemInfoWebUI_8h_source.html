25  void updateChartData() {
26  if (!sys) return;
27  const auto& metrics = sys->getMetrics();
28 
29  // Initialize with current values if not done yet
30  if (!chartInitialized) {
31  float heapPercent = metrics.totalHeap > 0 ? ((float)metrics.freeHeap / metrics.totalHeap) * 100.0f : 0.0f;
32  for (int i = 0; i < CHART_DATA_SIZE; i++) {
33  heapHistory[i] = heapPercent;
34  cpuHistory[i] = metrics.cpuLoad;
35  }
36  chartInitialized = true;
37  return;
38  }
39 
40  // Throttle updates to the component interval
41  int interval = sys->getUpdateInterval();
42  if (millis() - lastChartUpdate < (unsigned long)interval) return;
43 
44  // Update with new values (scrolling buffer)
45  float heapPercent = metrics.totalHeap > 0 ? ((float)metrics.freeHeap / metrics.totalHeap) * 100.0f : 0.0f;
46  heapHistory[chartIndex] = heapPercent;
47  cpuHistory[chartIndex] = metrics.cpuLoad;
48 
49  // Move to next position (circular buffer)
50  chartIndex = (chartIndex + 1) % CHART_DATA_SIZE;
51  lastChartUpdate = millis();
52  }
