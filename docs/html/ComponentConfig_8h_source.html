303  ValidationResult validateIPAddress(const ConfigParam& param, const String& value) const {
304  // Simple IP address validation
305  int parts = 0;
306  int start = 0;
307  for (int i = 0; i <= value.length(); i++) {
308  if (i == value.length() || value[i] == '.') {
309  if (i == start) {
310  return ValidationResult(ComponentStatus::ConfigError,
311  "Invalid IP address format", param.name);
312  }
313  String part = value.substring(start, i);
314  int num = part.toInt();
315  if (num < 0 || num > 255) {
316  return ValidationResult(ComponentStatus::ConfigError,
317  "Invalid IP address range", param.name);
318  }
319  parts++;
320  start = i + 1;
321  }
322  }
323  if (parts != 4) {
324  return ValidationResult(ComponentStatus::ConfigError,
325  "Invalid IP address format", param.name);
326  }
327  return ValidationResult();
328  }
