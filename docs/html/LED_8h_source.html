386  void updateEffects() {
387  unsigned long currentTime = millis();
388 
389  for (size_t i = 0; i < ledStates.size(); i++) {
390  auto& state = ledStates[i];
391 
392  if (!state.enabled || state.effect == LEDEffect::Solid) {
393  if (state.effect == LEDEffect::Solid) {
394  setLEDOutput(i, state.currentColor, state.brightness);
395  }
396  continue;
397  }
398 
399  // Calculate effect phase (0.0 to 1.0)
400  float deltaTime = (currentTime - state.lastUpdate) / (float)state.effectSpeed;
401  state.effectPhase += deltaTime;
402 
403  if (state.effectPhase > 1.0) {
404  state.effectPhase = 0.0;
405  }
406 
407  state.lastUpdate = currentTime;
408 
409  // Apply effect
410  LEDColor outputColor = state.currentColor;
411  uint8_t outputBrightness = state.brightness;
412 
413  switch (state.effect) {
414  case LEDEffect::Blink:
415  outputBrightness = (state.effectPhase < 0.5) ? state.brightness : 0;
416  break;
417 
418  case LEDEffect::Fade:
419  outputBrightness = (uint8_t)(state.brightness * (sin(state.effectPhase * 2 * PI) + 1) / 2);
420  break;
421 
422  case LEDEffect::Pulse:
423  if (state.effectPhase < 0.3) {
424  outputBrightness = (uint8_t)(state.brightness * sin(state.effectPhase * PI / 0.3));
425  } else if (state.effectPhase < 0.5) {
426  outputBrightness = (uint8_t)(state.brightness * sin((state.effectPhase - 0.3) * PI / 0.2));
427  } else {
428  outputBrightness = 0;
429  }
430  break;
431 
432  case LEDEffect::Rainbow:
433  if (ledConfigs[i].isRGB) {
434  float hue = state.effectPhase * 360.0;
435  // Simple HSV to RGB conversion
436  if (hue < 120) {
437  outputColor = LEDColor(255 - hue * 2.125, hue * 2.125, 0);
438  } else if (hue < 240) {
439  outputColor = LEDColor(0, 255 - (hue - 120) * 2.125, (hue - 120) * 2.125);
440  } else {
441  outputColor = LEDColor((hue - 240) * 2.125, 0, 255 - (hue - 240) * 2.125);
442  }
443  outputBrightness = state.brightness;
444  }
445  break;
446 
447  case LEDEffect::Breathing:
448  outputBrightness = (uint8_t)(state.brightness * (1 - cos(state.effectPhase * 2 * PI)) / 2);
449  break;
450 
451  default:
452  break;
453  }
454 
455  setLEDOutput(i, outputColor, outputBrightness);
456  }
