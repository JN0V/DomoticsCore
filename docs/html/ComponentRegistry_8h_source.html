255  initializationOrder.clear();

257  std::map<String, std::vector<String>> dependents;
258 
259  // Initialize in-degree count
260  for (const auto& comp : components) {

262  inDegree[name] = 0;
263  dependents[name] = {};
264  }
265 
266  // Build dependency graph
267  for (const auto& comp : components) {
268  String name = comp->getName();
269  auto deps = comp->getDependencies();
270 
271  for (const String& dep : deps) {
272  if (componentMap.find(dep) == componentMap.end()) {
273  DLOG_E(LOG_CORE, "Component '%s' depends on unregistered component '%s'",
274  name.c_str(), dep.c_str());
275  return false;
276  }
277 
278  dependents[dep].push_back(name);
279  inDegree[name]++;
280  }
281  }
282 
283  // Topological sort using Kahn's algorithm
284  std::vector<String> queue;
285  for (const auto& pair : inDegree) {
286  if (pair.second == 0) {
287  queue.push_back(pair.first);
288  }
289  }
290 
291  while (!queue.empty()) {
292  String current = queue.back();
293  queue.pop_back();
294

296

298  inDegree[dependent]--;
299  if (inDegree[dependent] == 0) {

301  }
302  }
303  }
304
